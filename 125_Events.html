    Hændelser

    Om modulet

    Dette model handler om hændelser (events). Det er muligheden for at objekter kan afvikle en eller flere metoder når en given hændelse sker. Det vigtigste at forstå er princippet og formålet med hændelser så vi starter med at se på nogle eksempler. Herefter ser vi på hvordan du selv kan skabe klasser med hændelser.

    Grundlæggende om hændelser

    Video Grundlæggende om hændelser (10 min.)

    Se video

    Hændelser giver mulighed for at objekter kan afvikle kode når “noget” sker - det kunne være

    • Når der er gået et specielt tidsrum
    • Når der oprettes eller slettes en fil fra en folder
    • Når en konto går i minus
    • Når der slås en sekser med en terning

    Eksempel - en timer

    Prøv at kopiere følgende kode til en tom konsolapplikation:

    using System;

namespace Demo
{
  class Program
  {
    static void Main(string[] args)
    {

      System.Timers.Timer t = new System.Timers.Timer();
      t.Interval = 1000;
      t.Enabled = true;
      t.Elapsed += (s, e) => Console.WriteLine("tiktak");
      do { } while (true);

    }
  }
}

    Bemærk, at der hvert sekund udskrives “tiktak”, og at t.Elapsed += er brug af delegates jf. tidligere.

    Eksempel - Hold øje med en mappe

    Prøv følgende kode:

    using System;

namespace Demo
{
    class Program
    {
        static void Main(string[] args)
        {
            System.IO.FileSystemWatcher w = new System.IO.FileSystemWatcher(@"c:\tmp");
            w.EnableRaisingEvents = true;
            w.Created += (s, e) => Console.WriteLine("Fil oprettet");
            w.Deleted += (s, e) => Console.WriteLine("Fil slettet");
            do { } while (true);
        }
    }
}

    Bemærk, at der udskrives på konsol når der oprettes og slettes en fil i c:\tmp.

    Eksempel - Når en konto går i minus

    Prøv følgende:

    using System;

namespace Demo
{
    class Program
    {
        static void Main(string[] args)
        {
            Konto k = new Konto(100);
            k.KontoIMinus += (s, e) => Console.WriteLine("MINUS..");
            k.Indsæt(10);
            Console.WriteLine(k.Saldo);
            k.Udtræk(50);
            Console.WriteLine(k.Saldo);
            k.Udtræk(10);
            Console.WriteLine(k.Saldo);
            k.Udtræk(60);
            Console.WriteLine(k.Saldo);

        }
    }

    class Konto
    {
        public double Saldo { get; private set; }
        public event EventHandler KontoIMinus;
        public Konto(double saldo)
        {
            Saldo = saldo;
        }

        public void Indsæt(double beløb)
        {
            Saldo += beløb;
            if (Saldo < 0)
                KontoIMinus?.Invoke(this, new EventArgs());
        }

        public void Udtræk(double beløb)
        {
            Saldo -= beløb;
            if (Saldo < 0)
                KontoIMinus?.Invoke(this, new EventArgs());
        }
    }
}

    Bemærk, at Konto-klassen giver mulighed for at afvikle noget kode når saldoen er under nul.

    Eksempel - når der slås en sekser

    Prøv følgende:

    using System;

namespace Demo
{
    class Program
    {
        static void Main(string[] args)
        {
            Terning t = new Terning();
            t.ErSekser += (s, e) =>
            {
                Console.WriteLine("SEKSER!!!!");
                Console.Beep();
            };

            for (int i = 0; i < 10; i++)
            {
                t.Ryst();
                t.Skriv();
            }

        }
    }

    public class Terning
    {
        private System.Random rnd = new Random();
        public event EventHandler ErSekser;

        public int Værdi { get; private set; }

        public Terning()
        {
            this.Ryst();
        }

        public void Ryst()
        {
            this.Værdi = rnd.Next(1, 7);
            if (this.Værdi == 6)
            {
                ErSekser?.Invoke(this, new EventArgs());
            }
        }

        public void Skriv()
        {
            Console.WriteLine($"[{this.Værdi}]");
        }
    }
}

    Hvad er en hændelse

    Video Hvad er en hændelse (6 min.)

    Se video

    Egentlig er en hændelse (event) i bund og grund en delegate som tidligere gennemgået, men er implementeret som en hændelse primært fordi det skaber en tydelig adskillelse mellem de andre medlemstyper:

    • Felter
    • Egenskaber
    • Metoder
    • Hændelser

    Yderligere er en hændelse beskyttet af runtime på forskellige måder, og det er mulighed at skabe indkapsling med add/remove (i stedet for get/set).

    Forskellen på en “rå” delegate og en event

    using System;

namespace Demo
{
    class Program
    {
        static void Main(string[] args)
        {
            EventDelegate ed = new EventDelegate();
            ed.DelegateTest += (s, e) => Console.WriteLine("delegate");
            ed.EventTest += (s, e) => Console.WriteLine("event");
            ed.Test();

            ed.DelegateTest = null; // Det er ok
            // ed.EventTest = null; // Det må man ikke (man kunne jo ødelægge andres abonnering på hændelsen)
        }
    }

    class EventDelegate
    {
        public Action<object, EventArgs> DelegateTest;
        public event Action<object, EventArgs> EventTest;

        public void Test() {
            DelegateTest?.Invoke(this, new EventArgs());
            EventTest?.Invoke(this, new EventArgs());
        }
    }
}

    Bemærk, at der ikke er meget forskel - men event er en separat medlemstype og er beskyttet af runtime (på flere måder).

    event-kodeordet

    Video event-kodeordet (4 min.)

    Se video

    Brug event-kodeordet til at definere en hændelse (delegate)

    public class EventDemo
{
    public event Action Event1;                     // bare en void
    public event Action<string> Event2;             // void(string)
    public event Action<object, EventArgs> Event3;  // void(object, EventArgs)
    public event EventHandler Event4;               // void(object, EventArgs)
}

    De fleste benytter Microsoft’s System.EventHandler (object, EventArgs) som predefineret delegate:

    public event EventHandler Event1;

    men det er helt op til dig.

    Afvikling af metoder

    Når man skal afvikle en hændelse i klassen foregår det på samme måde som ved delegates (for… det er jo en delegate). Man benytter typisk Invoke-metoden for nemt at kunne checke for null (se tidligere modul):

    class Konto
{
    public double Saldo { get; private set; }
    public event EventHandler KontoIMinus;
    public Konto(double saldo)
    {
        Saldo = saldo;
    }

    public void Indsæt(double beløb)
    {
        Saldo += beløb;
        if (Saldo < 0)
            KontoIMinus?.Invoke(this, new EventArgs());
    }

    public void Udtræk(double beløb)
    {
        Saldo -= beløb;
        if (Saldo < 0)
            KontoIMinus?.Invoke(this, new EventArgs());
    }
}

    Hvis man benytter den indbyggede delegate EventHandler er det første argument typisk det objekt der smider hændelsen, og det andet et tomt objekt af klassen EventArgs eller en af nedarvet klasse.

    Din egen EventArgs-klasse

    Video Din egen EventArgs-klasse (5 min.)

    Se video

    Du kan eventuelt skabe din egen eventargs-klasse ved at arve fra EventArgs, og så benytte den generiske EventHandler.

    Bemærk forskellen på konto1 og konto2 - både i definition og brug:

    using System;

namespace Demo
{
    class Program
    {
        static void Main(string[] args)
        {
            Konto1 k1 = new Konto1(10);
            k1.KontoIMinus += (s, e) => Console.WriteLine("MINUS");
            Konto2 k2 = new Konto2(10);
            k2.KontoIMinus += (s, e) => Console.WriteLine($"MINUS med {e.Saldo}");
            k1.Udtræk(20);
            k2.Udtræk(20);
        }
    }

    class Konto1
    {
        public double Saldo { get; private set; }
        public event EventHandler KontoIMinus;
        public Konto1(double saldo)
        {
            Saldo = saldo;
        }

        public void Indsæt(double beløb)
        {
            Saldo += beløb;
            if (Saldo < 0)
                KontoIMinus?.Invoke(this, new EventArgs());
        }

        public void Udtræk(double beløb)
        {
            Saldo -= beløb;
            if (Saldo < 0)
                KontoIMinus?.Invoke(this, new EventArgs());
        }
    }


    class Konto2
    {
        public double Saldo { get; private set; }
        public event EventHandler<MineEventArgs> KontoIMinus;
        public Konto2(double saldo)
        {
            Saldo = saldo;
        }

        public void Indsæt(double beløb)
        {
            Saldo += beløb;
            if (Saldo < 0)
                KontoIMinus?.Invoke(this, new MineEventArgs { Saldo = this.Saldo });
        }

        public void Udtræk(double beløb)
        {
            Saldo -= beløb;
            if (Saldo < 0)
                KontoIMinus?.Invoke(this, new MineEventArgs { Saldo = this.Saldo });
        }
    }

    public class MineEventArgs : EventArgs
    {
        public double Saldo { get; set; }
    }

}
