    Generics

    • Der er en masse indbyggede typer og metoder der benytter generics
    • System.Collections.Genetic
    • LINQ
    • Meget mere
    • Generics er overordet en template for en type der erstattes på compile time
    • Typisk set som typen T
    • Slipper for at typekonvertere til og fra Object
    • Kan oprette egne interfaces, klasser, metoder, hændelser og delegates

    Udvikling af egne metoder

    • Klassiske eksempler
    int a = 10;
int b = 20;
// Behøver ikke Swap<int> - Swap() er ok
Swap<int>(ref a, ref b);
Console.WriteLine($"a={a} b={b}");

string c = "a";
string d = "b";
// Behøver ikke Swap<string>
Swap(ref c, ref d);
Console.WriteLine($"c={c} d={d}");

Faktura e = new Faktura() { Id = 1 };
Faktura f = new Faktura() { Id = 2 };
Swap(ref e, ref f);
Console.WriteLine($"e={e.Id} f={f.Id}");

void Swap<T>(ref T t1, ref T t2)
{
    T tmp = t1;
    t1 = t2;
    t2 = tmp;
}

var lst1 = new List<int>();
AddToCollection<int>(1, lst1);
AddToCollection(1, lst1);
Console.WriteLine($"Elementer i lst1; {lst1.Count}");

var lst2 = new List<Faktura>();
AddToCollection<Faktura>(new Faktura(), lst2);
AddToCollection<Faktura>(new Faktura(), lst2);
AddToCollection<Faktura>(new Faktura(), lst2);
Console.WriteLine($"Elementer i lst2; {lst2.Count}");

void AddToCollection<T>(T item, List<T> collection) 
{
    // log+sikkerhed
    collection.Add(item);
}

    Constrains og default

    • Kan styre typen T på forskellig vis

    • where T : struct

    • where T : class
    • where T : unmanaged
    • where T : new() (skal have en default constructor)
    • where T : \<base class>

    • where T : \<interface>

    • Kodeordet default kan finde default værdien

    var lst1 = new List<Faktura>();
AddToCollection<Faktura>(new Faktura(), lst1);
AddToCollection<Faktura>(new Faktura(), lst1);
AddToCollection<Faktura>(new Faktura(), lst1);

AddToCollection2<Faktura>(new Faktura(), lst1);

AddToCollection3<Faktura>(new Faktura(), lst1);

Console.WriteLine($"Elementer i lst1; {lst1.Count}");

var lst2 = new List<int>();
//AddToCollection<Faktura>(2, lst2);  // fejl


void AddToCollection<T>(T item, List<T> collection)  where T : Faktura
{
    // log+sikkerhed
    collection.Add(item);
}

void AddToCollection2<T>(T item, List<T> collection) where T : class
{
    // log+sikkerhed
    collection.Add(item);
}

// Skal have en default constructor
void AddToCollection3<T>(T item, List<T> collection) where T : new()
{
    // log+sikkerhed
    collection.Add(item);
}

int[] lst3 = { 1, 2, 8, 1, 7, 1, 12, 3, 6 };
int res = FindStørsteVærdi<int>(lst3);

string[] lst4 = { "a", "y", "b" };

T FindStørsteVærdi<T>(T[] andre) where T : IComparable
{
    // Bemærk - default værdi
    T i = default(T);
    Console.WriteLine($"Default værdi: {i}");
    foreach (var item in andre)
    {
        if (item.CompareTo(i) > 0)
            i = item;
    }
    return i;
}

    Generiske typer

    • Du kan også skabe egne generiske strukturer og klasser
    using System;
using System.Collections.Generic;
using System.Text;

namespace GenericsDemo
{
    struct Point<T>
    {
        T x;
        T y;

        public Point(T x, T y) {
            this.x = x;
            this.y = y;
        }

        public T X { get { return this.x; } }
        public T Y { get { return this.y; } }
    }

    class Garage<T> where T : Bil {
        private List<T> biler = new List<T>();

        public void SætBilIGarage(T bil) {
            biler.Add(bil);
        }

        public List<T> HentBiler() {
            return this.biler;
        }
    }

    class Bil {
        public string Model { get; set; }
    }

    class Sportvogn : Bil 
    {
        // ...
    }
}

    GenericsDemo.Point<int> p1 = new GenericsDemo.Point<int>(1, 2);
GenericsDemo.Point<double> p2 = new GenericsDemo.Point<double>(1.3, 2.4);

GenericsDemo.Garage<GenericsDemo.Bil> garage = new GenericsDemo.Garage<GenericsDemo.Bil>();
GenericsDemo.Sportvogn b1 = new GenericsDemo.Sportvogn();
garage.SætBilIGarage(b1);
