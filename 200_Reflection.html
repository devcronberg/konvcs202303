    Assembly

    • Find typer
    • Oprettelse instanser
    • Afvikling af metoder
    namespace Reflection1
{
    public class TestKlasse
    {
        public int Id;
        public string Navn { get; set; }
        public void Test() { }
        public int Test1() => this.Id;
        public string Test2(int a, char b) => this.Navn?.ToUpper() + new string(b, a).ToUpper();
        public static void Test3() { }
        public TestKlasse() { }

        public TestKlasse(int id, string navn)
        {
            this.Id = id;
            this.Navn = navn;
        }
    }
}

    // using System.Reflection;
Assembly assem = typeof(Reflection1.TestKlasse).Assembly;

Console.WriteLine("Assembly Full Name:");
Console.WriteLine(assem.FullName);

AssemblyName assemName = assem.GetName();
Console.WriteLine("\nName: {0}", assemName.Name);
Console.WriteLine("Version: {0}.{1}",
    assemName.Version.Major, assemName.Version.Minor);

Object o = assem.CreateInstance("Reflection1.TestKlasse", false,
    BindingFlags.ExactBinding,
    null, new Object[] { 1, "a" }, null, null);

MethodInfo m = assem.GetType("Reflection1.TestKlasse").GetMethod("Test1");
Object ret = m.Invoke(o, new Object[] { });
Console.WriteLine($"Test1: {ret}");

m = assem.GetType("Reflection1.TestKlasse").GetMethod("Test2");
ret = m.Invoke(o, new Object[] { 3, 'x' });
Console.WriteLine($"Test2: {ret}");


/*
 ---------- Output: ----------

Eksempler, Version=1.0.0.0, Culture=neutral, PublicKeyToken=null

Version: 1.0
Test1: 1
Test2: AXXX

*/

    Attributter

    • []-format
    • Ekstra metadata om typer og medlemmer
    • Bruges meget af frameworket
      • [Obsolete]
      • [AssemblyVersion]
      • [Serializable]
      • ..
    • Nemt at lave sine egne
    using System;

namespace Reflection2
{
    [MinAttribut1]
    public class MinKlasse {

        [MinAttribut2(1, 2)]
        public void Test() { }
    }


    public class MinAttribut1 : Attribute
    {
    }

    public class MinAttribut2 : Attribute
    {

        private int a;
        private int b;

        public MinAttribut2()
        {

        }

        public MinAttribut2(int a=0, int b=1)
        {
            this.a = a;
            this.b = b;
        }
    }
}
    • Attributter kan aflæses gennem reflection
    Reflection2.MinKlasse o = new Reflection2.MinKlasse();

foreach (var item in o.GetType().GetCustomAttributes())
    Console.WriteLine(item.ToString());

foreach (var m in o.GetType().GetMethods())
    foreach (var a in m.GetCustomAttributes())
        Console.WriteLine(a.ToString());

/*
 ---------- Output: ----------

Reflection2.MinAttribut1
Reflection2.MinAttribut2

*/

    Automatisk kodegenerering

    • Nogen gange giver det mening at genere kode dynamisk og automatisk
      • især ORM-produkter som EF ol. men der er mange andre eksempler
    • Frameworket har System.CodeDom som giver mulighed for at genere source (VB, C#) og eksekverbar IL
    • Man opbygger et træ af objekter som repræsenterer kode, og evt kompilere dette til sidst
    • Kort eksempel som genererer en enum ud fra en database
    // using System.CodeDom
// using Microsoft.CSharp;

// Antag at dette er hentet fra et database-kald
// og skal ligge til grund for en enum

Dictionary<int, string> StatusKoder = new Dictionary<int, string>();
StatusKoder.Add(0, "IkkeAngivet");
StatusKoder.Add(1, "Ligegyldigt");
StatusKoder.Add(2, "IkkeVigtigt");
StatusKoder.Add(4, "Vigtigt");
StatusKoder.Add(8, "MegetVigtig");

// Automatisk generering af enum
CodeCompileUnit unit = new CodeCompileUnit();

CodeNamespace ns = new CodeNamespace("MitNamespace");
ns.Imports.Add(new CodeNamespaceImport("System"));
unit.Namespaces.Add(ns);

CodeTypeDeclaration en = new CodeTypeDeclaration("StatusKoder");
en.IsEnum = true;
ns.Types.Add(en);

foreach (var statuskode in StatusKoder)
{
    CodeMemberField f = new CodeMemberField("", statuskode.Value);
    f.InitExpression = new CodePrimitiveExpression(statuskode.Key);
    en.Members.Add(f);
}

CSharpCodeProvider p = new CSharpCodeProvider();
using (System.IO.TextWriter w = System.IO.File.CreateText(@"c:\temp\statuskoder.cs"))
{
    p.GenerateCodeFromCompileUnit(unit, w, new System.CodeDom.Compiler.CodeGeneratorOptions() {  BlankLinesBetweenMembers = false});
}

// Følgende indhold vil nu være at finde i c:\temp\statuskoder.cs
// (Bemærk - sidste komma i en enum er lovligt i C#)
Console.WriteLine(System.IO.File.ReadAllText(@"c:\temp\statuskoder.cs"));



/*
 ---------- Output: ----------

//------------------------------------------------------------------------------
// <auto-generated>
//     This code was generated by a tool.
//     Runtime Version:4.0.30319.42000
//
//     Changes to this file may cause incorrect behavior and will be lost if
//     the code is regenerated.
// </auto-generated>
//------------------------------------------------------------------------------

namespace MitNamespace {
    using System;

    public enum StatusKoder {
        IkkeAngivet = 0,
        Ligegyldigt = 1,
        IkkeVigtigt = 2,
        Vigtigt = 4,
        MegetVigtig = 8,
    }
}


*/

    Der er sågar mulighed for at skabe kode dynamisk og afvikle denne med det samme!!

    Scripting

    Brug af Roslyn (open-source C# compilers with rich code analysis APIs) til at generere udtryk fra en streng (eval() i C#)

    // Nuget Add Package Microsoft.CodeAnalysis.Scripting
using Microsoft.CodeAnalysis.CSharp.Scripting;
using Microsoft.CodeAnalysis.Scripting;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;

namespace ConsoleApp2
{
    internal class Program
    {
        static async Task Main(string[] args)
        {

            var albums = new List<Album>
            {
                new Album { Quantity = 10, Artist = "A", Title = "T1" },
                new Album { Quantity = 50, Artist = "B", Title = "T2" },
                new Album { Quantity = 200, Artist = "C", Title = "T3" },
            };

            string discountFilter = "album => album.Quantity > 10";
            ScriptOptions options = ScriptOptions.Default.AddReferences(typeof(Album).Assembly);
            Func<Album, bool> discountFilterExpression = await CSharpScript.EvaluateAsync<Func<Album, bool>>(discountFilter, options);
            List<Album> res = albums.Where(discountFilterExpression).ToList();
            foreach (var item in res)
                Console.WriteLine(item.Title);
        }
    }

    public class Album
    {
        public int Quantity { get; set; }
        public string Artist { get; set; }
        public string Title { get; set; }
    }
}
