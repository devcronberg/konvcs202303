    Optimering

    Om modulet

    I dette modul ser vi på optimering af kode - herunder brug af IDisposable.

    Objektets liv og død

    • Objekt oprettes
    • constructor køres
    • Objektet lever sit liv på heap’en
    • Objektet slettes (enten ved null eller scope)
    • Finalize/destructor
    • Hukommelse frigives
    • runtime håndterer oprettelse af objektet
    • Garbage Collection rydder op

    Tilgange til garbage Collection

    • System.GC
    • Collect();
    • WaitForPendingFinalizers()
    • masse metoder men i praksis skal man lave GC gøre det automatisk

    Destructors

    • Finalize kommer fra System.Object og kan ikke overskrives
    • Brug en destructor
    • ~Navn

    Memory leaks

    • Man kan godt skabe mem.leaks i C#

    • Især events kan gøre ondt

    • Se (og kør i VS - se hukommelsesforbrug stige) dette eksempel hvor en App skaber objekter der binder en metode til “mor’s” event og glemmer at slette referencen
    using System;
using System.Collections.Generic;

namespace MemLeak1
{
    class Program
    {
        static void Main(string[] args)
        {
            MinApp app = new MinApp();

            for (int x = 0; x < 100; x++)
            {
                for (int i = 0; i < 10000; i++)
                {
                    app.OpretObjekt();
                }
                System.Threading.Thread.Sleep(200);
            }
        }
    }

    class MinApp
    {
        public event EventHandler Hjerteslag;
        public void OpretObjekt()
        {
            var o = new MitObjekt(this);
            o = null;
        }

        public void Tick()
        {
            if (Hjerteslag != null)
                Hjerteslag(this, new EventArgs());
        }
    }

    class MitObjekt
    {

        private MinApp mor;
        public MitObjekt(MinApp mor)
        {
            this.mor = mor;
            this.mor.Hjerteslag += SkrivTick;
        }

        private void SkrivTick(object s, EventArgs e)
        {
            Console.WriteLine("Tick");
        }

        ~MitObjekt()
        {
            this.mor.Hjerteslag -= SkrivTick;
        }

    }
}
    • Her bliver der ryddet op

using System.Collections.Generic;

namespace MemLeak2

    {

        public List<MitObjekt> lst = new List<MitObjekt>();

            if (Hjerteslag != null)

        public void FjernReferencer()
        {
            foreach (var item in lst)
            {
                item.FjernEvent();
            }
        }

    }

    {

        private MinApp mor;
        public MitObjekt(MinApp mor)
        {
            this.mor = mor;
            this.mor.Hjerteslag += SkrivTick;
        }

        private void SkrivTick(object s, EventArgs e)
        {
            Console.WriteLine("Tick");
        }

        public void FjernEvent()
        {
            this.mor.Hjerteslag -= SkrivTick;
        }

    }
}

    ref local

    Fra C# 7 kan metoder nu returnerer værdibaserede variabler by ref, og de skal gemmes i en “ref local” variabel.

    char[] arr = { 'r', 'e', 't'};

Console.WriteLine(string.Join(" ", arr));
char res1 = getChar1(2, arr);
res1 = 'f';
Console.WriteLine(string.Join(" ", arr));

ref char res2 = ref getChar2(2, arr);
res2 = 'f';
Console.WriteLine(string.Join(" ", arr));

char getChar1(int index, char[] a) {
    return a[index];
}
ref char getChar2(int index, char[] a)
{
    return ref a[index];
}

    Span

    I C# 7 blev Span introduceret og giver typesikker adgang til et område i hukommelsen (stack/heap). Se det som et vindue mod hukommelse med mulighed for at tilrette direkte.

    Bemærk - der er typisk tale om mikrooptimering!

    Eksempel på Span

    string[] array = { "a", "b", "c", "d", "e" };
// Slizing
var firstView = new Span<string>(array, 0, 3);
var secondView = new Span<string>(array, 2, 3);

firstView[0] = "w";
Console.WriteLine(string.Join(' ', array));
// array = { "w", "b", "c", "d", "e" }
firstView[2] = "x";
Console.WriteLine(string.Join(' ', array));
// array = { "w", "b", "x", "d", "e" }
secondView[0] = "y";
Console.WriteLine(string.Join(' ', array));
// array = { "w", "b", "y", "d", "e" }

    Span har en søster kaldet ReadOnlySpan som giver en readonly adgang.

    Eksempel på ReadOnlySpan

    {

    string navn = "Michell Cronberg";
    int mellemrum = navn.IndexOf(" ");
    string fornavn = navn.Substring(0, mellemrum);
    string efternavn = navn.Substring(mellemrum + 1);
    Console.WriteLine(fornavn);
    Console.WriteLine(efternavn);
    // Problem: fornavn og efternavn allokeres på heap (og efterfølgende skal fjernes)
}

{
    string navn = "Michell Cronberg";
    int mellemrum = navn.IndexOf(" ");
    ReadOnlySpan<char> span = navn.AsSpan();
    // 'slize' til et nyt vindue
    ReadOnlySpan<char> fornavn = span.Slice(0, mellemrum);
    ReadOnlySpan<char> efternavn = span.Slice(mellemrum + 1);
    Skriv(fornavn);
    Console.WriteLine();
    Skriv(efternavn);

    // Console.WriteLine mangler et overload der tager Span<T>
    void Skriv(ReadOnlySpan<char> chars)
    {
        for (int i = 0; i < chars.Length; i++)
            Console.Write(chars[i]);
    }


}

    Her er et mere grundigt eksempel: 

    // Kræver Benchmark.NET
// Eksempel kraftigt inspireret af 
// https://www.stevejgordon.co.uk/an-introduction-to-optimising-code-using-span-t

using BenchmarkDotNet.Attributes;
using BenchmarkDotNet.Order;
using BenchmarkDotNet.Running;

namespace Test
{
    class Program
    {
        static void Main(string[] args)
        {
            string name = "Michel A Cronberg";
            NameParser p = new NameParser();
            Console.WriteLine(p.GetLastNameUsingSplit(name));
            Console.WriteLine(p.GetLastNameUsingSubstring(name));
            Console.WriteLine(p.GetLastNameWithSpan(name));
            var result = BenchmarkRunner.Run(typeof(NameParser).Assembly);
        }
    }
    public class NameParser
    {
        public string GetLastNameUsingSplit(string fullName)
        {
            var s = fullName.Split(' ');
            return s.Length > -1
                ? s[s.Length - 1]
                : string.Empty;
        }

        public string GetLastNameUsingSubstring(string fullName)
        {
            var lastSpaceIndex = fullName.LastIndexOf(" ", StringComparison.Ordinal);

            return lastSpaceIndex == -1
                ? string.Empty
                : fullName.Substring(lastSpaceIndex + 1);
        }

        public string GetLastNameWithSpan(ReadOnlySpan<char> fullName)
        {
            var lastSpaceIndex = fullName.LastIndexOf(' ');

            // koster 40b at returnere en string - kunne returnere en readonlyspan<t>
            return lastSpaceIndex == -1
                ? string.Empty
                : fullName.Slice(lastSpaceIndex + 1).ToString();
        }
    }

    [RankColumn]
    [Orderer(SummaryOrderPolicy.FastestToSlowest)]
    [MemoryDiagnoser]
    public class NameParserBenchmarks
    {
        private const string FullName = "Michell A Cronberg";
        private static readonly NameParser Parser = new NameParser();

        [Benchmark(Baseline = true)]
        public void GetLastNameUsingSplit()
        {
            Parser.GetLastNameUsingSplit(FullName);
        }

        [Benchmark]
        public void GetLastNameUsingSubstring()
        {
            Parser.GetLastNameUsingSubstring(FullName);
        }

        [Benchmark]
        public void GetLastNameWithSpan()
        {
            Parser.GetLastNameWithSpan(FullName);
        }
    }
}

    Memory

    Da Span arbejder på stack’en kan den ikke bruges som felter i klasser, og dermed ikke i lambda eller som parametre i async metoder. Derfor findes Memory (ReadOnlyMemory) som giver disse muligheder:

    // async Task Test(Span<int> mem) --- fejl
async Task Test(Memory<int> mem)

}
