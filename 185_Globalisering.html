    Globalisering

    Om modulet

    Dette modul vil se på hvordan du kan skabe applikationer som arbejder med flere sprog.

    Globalisering og lokalisering

    • Globalisering er formatering af tal og datoer og sammlingning af strenge

    • Lokalisering er brug af ressource filer til oversættelse

    • Ikke dækket i dette modul

    Culture

    • En Culture består af

    • Land (da, en, de)

    • Kultur (DK, US, DE)
    • Tilsammen danner de en specifik kultur (da-DK, en-US)
    • Et rent land kaldes en neutral kultur (da, en, de)
    • Brug CultureInfo til at repræsentere en kultur
    System.Globalization.CultureInfo ci;
ci = new System.Globalization.CultureInfo("da-DK");
Console.WriteLine($"{ci.Name} {ci.NativeName} {ci.DisplayName}");
ci = new System.Globalization.CultureInfo("en-US");
Console.WriteLine($"{ci.Name} {ci.NativeName} {ci.DisplayName}");
ci = new System.Globalization.CultureInfo("en");
Console.WriteLine($"{ci.Name} {ci.NativeName} {ci.DisplayName}");

ci = System.Globalization.CultureInfo.CurrentCulture;
Console.WriteLine($"{ci.Name} {ci.NativeName} {ci.DisplayName}");

// Mange måder at finde culture på - brug evt kald til framework
var lst = System.Globalization.CultureInfo.GetCultures(System.Globalization.CultureTypes.NeutralCultures);
string tmp = string.Join(" ", lst.Skip(51).Take(25).Select(i => i.DisplayName));
Console.WriteLine(tmp);


/*
 ---------- Output: ----------

da-DK dansk (Danmark) Dansk (Danmark)
en-US English (United States) Engelsk (USA)
en English Engelsk
da-DK dansk (Danmark) Dansk (Danmark)
Græsk Engelsk esperanto Spansk Estisk Baskisk ewondo Persisk Fulah Fulah Finsk Filippinsk Færøsk Fransk furlan Vestfrisisk Irsk Skotsk gælisk Galisisk Guarani Tysk (Schweiz) Gujarati Ekegusii Gaelg Hausa

*/
    • RegionInfo giver flere informationer om landet - herunder valuta
    System.Globalization.CultureInfo ci;
ci = System.Globalization.CultureInfo.CurrentCulture;
System.Globalization.RegionInfo ri;
ri = new System.Globalization.RegionInfo(ci.Name);

Console.WriteLine($"Valuta {ri.CurrencyNativeName} {ri.CurrencyEnglishName}");
Console.WriteLine($"Navn {ri.DisplayName} {ri.EnglishName}");


/*
 ---------- Output: ----------

Valuta Dansk krone Danish Krone
Navn Danmark Denmark

*/

    Formatering og typekonvertering

    • Brug kultur til formatering
    • Brug kultur til typekonvertering
    System.Globalization.CultureInfo dk;
dk = new System.Globalization.CultureInfo("da-DK");
System.Globalization.CultureInfo us;
us = new System.Globalization.CultureInfo("en-US");

double tal = 1234567.433456;
DateTime dato = new DateTime(2018, 1, 1);

Console.WriteLine($"da: {tal.ToString(dk)}");
Console.WriteLine($"us: {tal.ToString(us)}");
Console.WriteLine($"da: {tal.ToString("N2", dk)}");
Console.WriteLine($"us: {tal.ToString("N2", us)}");
Console.WriteLine($"da: {tal.ToString("C2", dk)}");
Console.WriteLine($"us: {tal.ToString("C2", us)}");

Console.WriteLine($"da: {dato.ToString("d", dk)}");
Console.WriteLine($"us: {dato.ToString("d", us)}");
Console.WriteLine($"da: {dato.ToString("F", dk)}");
Console.WriteLine($"us: {dato.ToString("F", us)}");

// Type konvertering
tal = System.Convert.ToDouble("23.434,43", dk);
Console.WriteLine($"tal: {tal.ToString(dk)}");
tal = System.Convert.ToDouble("23,434.43", us);
Console.WriteLine($"tal: {tal.ToString(dk)}");

dato = System.Convert.ToDateTime("30-1-2018", dk);
Console.WriteLine($"dato: {dato.ToString("d", dk)}");
dato = System.Convert.ToDateTime("2018-1-30", us);
Console.WriteLine($"dato: {dato.ToString("d", dk)}");


/*
 ---------- Output: ----------

da: 1234567,433456
us: 1234567.433456
da: 1.234.567,43
us: 1,234,567.43
da: 1.234.567,43 kr.
us: $1,234,567.43
da: 01-01-2018
us: 1/1/2018
da: 1. januar 2018 00:00:00
us: Monday, January 1, 2018 12:00:00 AM
tal: 23434,43
tal: 23434,43
dato: 30-01-2018
dato: 30-01-2018

*/

    Sammenligning og sortering

    System.Globalization.CultureInfo dk;
dk = new System.Globalization.CultureInfo("da-DK");

System.Globalization.CultureInfo us;
us = new System.Globalization.CultureInfo("en-US");

// Returnerer -1 0 1
Console.WriteLine(dk.CompareInfo.Compare("Aarhus", "Århus", System.Globalization.CompareOptions.IgnoreCase));
Console.WriteLine(us.CompareInfo.Compare("Aarhus", "Århus", System.Globalization.CompareOptions.IgnoreCase));

string[] lst = { "København", "Aalborg", "Århus", "Odense" };
var lst1 = lst.OrderBy(i => i, StringComparer.Create(dk, true));
var lst2 = lst.OrderBy(i => i, StringComparer.Create(us, true));

Console.WriteLine(string.Join(" ", lst1));
Console.WriteLine(string.Join(" ", lst2));


/*
 ---------- Output: ----------

1
-1
København Odense Aalborg Århus
Aalborg Århus København Odense

*/
